    W/A/S/D Movement

    Simple w/a/s/d movement. if you are wondering why i multiply deltatime by 200? I have no idea but it works so don't question it!

    InitWindow(800, 600, "W/A/S/D Movement")

local rectangle = CreateObject("Rectangle", 400, 300, 100, 100, 0)

local speed = 1

while not WindowShouldClose() do
    BeginDrawing()
    ClearBackground(0, 0, 0)

    local dtime = GetDeltaTime() * 200

    if Input.IsKeyDown(87) then
        rectangle:SetPosition(rectangle:GetPositionX(), rectangle:GetPositionY() - speed * dtime)
    end

    if Input.IsKeyDown(83) then
        rectangle:SetPosition(rectangle:GetPositionX(), rectangle:GetPositionY() + speed * dtime)
    end

    if Input.IsKeyDown(65) then
        rectangle:SetPosition(rectangle:GetPositionX() - speed * dtime, rectangle:GetPositionY())
    end

    if Input.IsKeyDown(68) then
        rectangle:SetPosition(rectangle:GetPositionX() + speed * dtime, rectangle:GetPositionY())
    end

    rectangle:Draw()

    EndDrawing()
end

    Mouse Follower

    A rectangle that follows you'r mouse

    InitWindow(800, 600, "Mouse Follower")

local rectangle = CreateObject("Rectangle", 400, 300, 100, 100, 0)

while not WindowShouldClose() do
    BeginDrawing()
    ClearBackground(0, 0, 0)

    rectangle:SetPosition(Input.GetMouseX(), Input.GetMouseY())
    rectangle:Draw()

    EndDrawing()
end

    Clicker game

    Yayyy, a clicker!

    InitWindow(800, 600, "Clicker Game")

local score = 0

local rectangle = CreateObject("Rectangle", 400, 300, 100, 100, 0)
local ScoreText = CreateObject("Text", 325, 150, 100, 100, 0)

ScoreText:SetCustomProperty("text", "Score: ".. score)
ScoreText:SetCustomProperty("fontSize", 40)

function AABB(ax, ay, aw, ah, bx, by, bw, bh)
    local rectX = bx - bw/2
    local rectY = by - bh/2

    return ax < rectX + bw and
           ax + aw > rectX and
           ay < rectY + bh and
           ay + ah > rectY
end

while not WindowShouldClose() do
    BeginDrawing()
    ClearBackground(0, 0, 0)

    local X = Input.MouseX()
    local Y = Input.MouseY()

    if Input.IsMouseButtonPressed(0) then
        if AABB(X, Y, 5, 5, 400, 300, 100, 100) then
            score = score + 1
        end
    end

    ScoreText:SetCustomProperty("text", "Score: ".. score)

    rectangle:Draw()
    ScoreText:Draw()
    EndDrawing()
end
